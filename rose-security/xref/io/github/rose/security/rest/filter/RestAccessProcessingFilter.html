16  package io.github.rose.security.rest.filter;
17  
18  import io.github.rose.security.rest.token.RestAccessAuthenticationToken;
19  import io.github.rose.security.support.DefaultTokenExtractor;
20  import io.github.rose.security.support.TokenExtractor;
21  import org.slf4j.Logger;
22  import org.slf4j.LoggerFactory;
23  import org.springframework.security.core.Authentication;
24  import org.springframework.security.core.AuthenticationException;
25  import org.springframework.security.core.context.SecurityContext;
26  import org.springframework.security.core.context.SecurityContextHolder;
27  import org.springframework.security.web.authentication.AbstractAuthenticationProcessingFilter;
28  import org.springframework.security.web.authentication.AuthenticationFailureHandler;
29  import org.springframework.security.web.util.matcher.RequestMatcher;
30  
31  import javax.servlet.FilterChain;
32  import javax.servlet.ServletException;
33  import javax.servlet.http.HttpServletRequest;
34  import javax.servlet.http.HttpServletResponse;
35  import java.io.IOException;
36  
37  public class RestAccessProcessingFilter extends AbstractAuthenticationProcessingFilter {
38      private static final Logger log = LoggerFactory.getLogger(RestAccessProcessingFilter.class);
39  
40      private final AuthenticationFailureHandler failureHandler;
41  
42      private final TokenExtractor tokenExtractor = new DefaultTokenExtractor();
43  
44      public RestAccessProcessingFilter(RequestMatcher matcher, AuthenticationFailureHandler failureHandler) {
45          super(matcher);
46          this.failureHandler = failureHandler;
47      }
48  
49      public TokenExtractor getTokenExtractor() {
50          return tokenExtractor;
51      }
52  
53      @Override
54      public Authentication attemptAuthentication(HttpServletRequest request, HttpServletResponse response)
55          throws AuthenticationException, IOException, ServletException {
56          RestAccessAuthenticationToken authenticationToken =
57              new RestAccessAuthenticationToken(tokenExtractor.extract(request));
58          authenticationToken.setDetails(authenticationDetailsSource.buildDetails(request));
59  
60          return getAuthenticationManager().authenticate(authenticationToken);
61      }
62  
63      @Override
64      protected void successfulAuthentication(
65          HttpServletRequest request, HttpServletResponse response, FilterChain chain, Authentication authResult)
66          throws IOException, ServletException {
67          SecurityContext context = SecurityContextHolder.createEmptyContext();
68          context.setAuthentication(authResult);
69          SecurityContextHolder.setContext(context);
70  
71          log.info("Login success: {}", authResult);
72  
73          chain.doFilter(request, response);
74      }
75  
76      @Override
77      protected void unsuccessfulAuthentication(
78          HttpServletRequest request, HttpServletResponse response, AuthenticationException failed)
79          throws IOException, ServletException {
80          SecurityContextHolder.clearContext();
81          failureHandler.onAuthenticationFailure(request, response, failed);
82      }
83  }
