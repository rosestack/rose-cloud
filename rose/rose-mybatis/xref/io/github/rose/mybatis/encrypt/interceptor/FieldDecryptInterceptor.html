16  package io.github.rose.mybatis.encrypt.interceptor;

18  import io.github.rose.mybatis.encrypt.FieldSetProperty;
19  import io.github.rose.mybatis.encrypt.IEncryptor;
20  import io.github.rose.mybatis.encrypt.IFieldBinder;
21  import io.github.rose.mybatis.encrypt.annotation.FieldBind;
22  import io.github.rose.mybatis.encrypt.annotation.FieldEncrypt;
23  import io.github.rose.mybatis.encrypt.util.FieldSetPropertyHelper;
24  import io.github.rose.mybatis.encrypt.util.InterceptorHelper;
25  import org.apache.ibatis.executor.resultset.ResultSetHandler;
26  import org.apache.ibatis.plugin.*;
27  import org.apache.ibatis.reflection.MetaObject;
28  import org.slf4j.Logger;
29  import org.slf4j.LoggerFactory;
30  
31  import java.sql.Statement;
32  import java.util.Properties;
33  
34  /**
35   * @author <a href="mailto:ichensoul@gmail.com">chensoul</a>
36   * @since 0.0.1
37   */
38  @Intercepts({
39      @Signature(
40          type = ResultSetHandler.class,
41          method = "handleResultSets",
42          args = {Statement.class})
43  })
44  public class FieldDecryptInterceptor implements Interceptor {
45      private static final Logger log = LoggerFactory.getLogger(FieldDecryptInterceptor.class);
46  
47      private final IEncryptor encryptor;
48  
49      private final IFieldBinder fieldBinder;
50  
51      private final String password;
52  
53      public FieldDecryptInterceptor(IEncryptor encryptor, IFieldBinder fieldBinder, String password) {
54          this.encryptor = encryptor;
55          this.fieldBinder = fieldBinder;
56          this.password = password;
57  
58          FieldSetPropertyHelper.init(null != encryptor, null != fieldBinder);
59      }
60  
61      public Object intercept(Invocation invocation) throws Throwable {
62          return InterceptorHelper.decrypt(invocation, (metaObject, fieldSetProperty) -> {
63              decrypt(this.encryptor, this.fieldBinder, this.password, metaObject, fieldSetProperty);
64          });
65      }
66  
67      public void decrypt(
68          IEncryptor encryptor,
69          IFieldBinder fieldBinder,
70          String password,
71          MetaObject metaObject,
72          FieldSetProperty fieldSetProperty) {
73          String fieldName = fieldSetProperty.getFieldName();
74          Object value = metaObject.getValue(fieldName);
75          if (null != value) {
76              if (null != encryptor && value instanceof String) {
77                  try {
78                      FieldEncrypt fieldEncrypt = fieldSetProperty.getFieldEncrypt();
79                      if (null != fieldEncrypt) {
80                          value = InterceptorHelper.getEncryptor(encryptor, fieldEncrypt.encryptor())
81                              .decrypt(fieldEncrypt.algorithm(), password, (String) value, null);
82                      }
83                  } catch (Exception e) {
84                      log.error("field decrypt", e);
85                  }
86              }
87  
88              if (null != fieldBinder) {
89                  FieldBind fieldBind = fieldSetProperty.getFieldBind();
90                  if (null != fieldBind) {
91                      fieldBinder.setMetaObject(fieldBind, value, metaObject);
92                  }
93              }
94              metaObject.setValue(fieldName, value);
95          }
96      }
97  
98      public Object plugin(Object var1) {
99          return var1 instanceof ResultSetHandler ? Plugin.wrap(var1, this) : var1;
100     }
101 
102     public void setProperties(Properties properties) {
103     }
104 }
