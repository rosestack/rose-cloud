27  import org.springframework.expression.Expression;
28  import org.springframework.expression.ParserContext;
29  import org.springframework.expression.common.TemplateParserContext;
30  import org.springframework.expression.spel.SpelCompilerMode;
31  import org.springframework.expression.spel.SpelParserConfiguration;

34

37  import java.time.ZoneId;
38  import java.time.ZonedDateTime;

41  import java.util.UUID;
42  import java.util.function.Function;
43  
44  /**
45   * @author <a href="mailto:ichensoul@gmail.com">chensoul</a>
46   * @since 0.0.1
47   */
48  public class SpringExpressionResolver implements Function<Object, Object> {
49      private static final Logger log = LoggerFactory.getLogger(SpringExpressionResolver.class);
50  
51      private static final ParserContext PARSER_CONTEXT = new TemplateParserContext("${", "}");
52      private static final ParameterNameDiscoverer PARAMETER_NAME_DISCOVERER = new DefaultParameterNameDiscoverer();
53      private static final SpelExpressionParser EXPRESSION_PARSER = new SpelExpressionParser(
54          new SpelParserConfiguration(SpelCompilerMode.IMMEDIATE, SpringExpressionResolver.class.getClassLoader()));
55      private static SpringExpressionResolver INSTANCE;
56      private final StandardEvaluationContext evaluationContext = new StandardEvaluationContext();
57  
58      protected SpringExpressionResolver() {
59          Properties properties = System.getProperties();
60          evaluationContext.setVariable("systemProperties", properties);
61          evaluationContext.setVariable("sysProps", properties);
62  
63          Map<String, String> environment = System.getenv();
64          evaluationContext.setVariable("environmentVars", environment);
65          evaluationContext.setVariable("environmentVariables", environment);
66          evaluationContext.setVariable("envVars", environment);
67          evaluationContext.setVariable("env", environment);
68  
69          evaluationContext.setVariable("tempDir", System.getProperty("java.io.tmpdir"));
70          evaluationContext.setVariable("zoneId", ZoneId.systemDefault().getId());
71  
72          evaluationContext.setBeanResolver(new BeanFactoryResolver(SpringContextHolder.getApplicationContext()));
73      }
74  
75      /**
76       * Gets instance of the resolver as a singleton.
77       *
78       * @return the instance
79       */
80      public static SpringExpressionResolver getInstance() {
81          if (INSTANCE == null) {
82              INSTANCE = new SpringExpressionResolver();
83          }
84          INSTANCE.initializeDynamicVariables();
85          return INSTANCE;
86      }
87  
88      /**
89       * Resolve string.
90       *
91       * @param value the value
92       * @return the string
93       */
94      public Object resolve(final String value) {
95          if (StringUtils.isNotBlank(value)) {
96              log.trace("Parsing expression as [{}]", value);
97              Expression expression = EXPRESSION_PARSER.parseExpression(value, PARSER_CONTEXT);
98              Object result = expression.getValue(evaluationContext);
99              log.trace("Parsed expression result is [{}]", result);
100             return result;
101         }
102         return value;
103     }
104 
105     @Override
106     public Object apply(final Object o) {
107         return resolve(o.toString());
108     }
109 
110     private void initializeDynamicVariables() {
111         evaluationContext.setVariable("randomNumber2", RandomStringUtils.randomNumeric(2));
112         evaluationContext.setVariable("randomNumber4", RandomStringUtils.randomNumeric(4));
113         evaluationContext.setVariable("randomNumber6", RandomStringUtils.randomNumeric(6));
114         evaluationContext.setVariable("randomNumber8", RandomStringUtils.randomNumeric(8));
115         evaluationContext.setVariable("randomString4", RandomStringUtils.randomAlphabetic(4));
116         evaluationContext.setVariable("randomString6", RandomStringUtils.randomAlphabetic(6));
117         evaluationContext.setVariable("randomString8", RandomStringUtils.randomAlphabetic(8));
118         evaluationContext.setVariable("uuid", UUID.randomUUID().toString());
119 
120         evaluationContext.setVariable("localStartWorkDay", DateUtils.getStartWorkDay().toString());
121         evaluationContext.setVariable("localEndWorkDay", DateUtils.getEndWorkDay().toString());
122         evaluationContext.setVariable("localStartDay", DateUtils.getStartDay().toString());
123         evaluationContext.setVariable("localEndDay", DateUtils.getEndDay().toString());
124         evaluationContext.setVariable("localDateTime", DateUtils.getLocalDateTime().toString());
125         evaluationContext.setVariable("localDateTimeUtc", DateUtils.getLocalDateTimeUTC().toString());
126         evaluationContext.setVariable("localDate", LocalDate.now(ZoneId.systemDefault()).toString());
127         evaluationContext.setVariable("localDateUtc", LocalDate.now(Clock.systemUTC()).toString());
128         evaluationContext.setVariable("zonedDateTime", ZonedDateTime.now(ZoneId.systemDefault()).toString());
129         evaluationContext.setVariable("zonedDateTimeUtc", ZonedDateTime.now(Clock.systemUTC()).toString());
130     }
