16  package io.github.rose.core.reflect;
17  
18  import io.github.rose.core.exception.RoseErrorCode;
19  import io.github.rose.core.exception.RoseException;
20  
21  import java.lang.reflect.AccessibleObject;
22  import java.lang.reflect.Field;
23  import java.lang.reflect.InvocationTargetException;
24  import java.lang.reflect.Method;
25  import java.security.AccessController;
26  import java.security.PrivilegedAction;
27  
28  public final class ReflectUtils {
29      private ReflectUtils() {
30          // no instantiation allowed
31      }
32  
33      /**
34       * Makes an {@link AccessibleObject} accessible by invoking
35       * {@link AccessibleObject#setAccessible(boolean)}
36       * in a {@link PrivilegedAction}.
37       *
38       * @param accessibleObject the object to make accessible.
39       * @param <T>              the type of object.
40       * @return the object made accessible.
41       */
42      public static <T extends AccessibleObject> T makeAccessible(T accessibleObject) {
43          AccessController.doPrivileged((PrivilegedAction<Object>) () -> {
44              accessibleObject.setAccessible(true);
45              return null;
46          });
47          return accessibleObject;
48      }
49  
50      /**
51       * Invokes the specified method while wrapping checked exception in a {@link RoseException}.
52       *
53       * @param method the method.
54       * @param self   the instance to invoke the method on.
55       * @param args   the method arguments.
56       * @param <T>    the type of the return value.
57       * @return the method returned value.
58       */
59      @SuppressWarnings("unchecked")
60      public static <T> T invoke(Method method, Object self, Object... args) {
61          try {
62              return (T) method.invoke(self, args);
63          } catch (IllegalAccessException | InvocationTargetException e) {
64              throw RoseException.wrap(e, RoseErrorCode.UNABLE_TO_INVOKE_METHOD)
65                  .put("method", method.toGenericString());
66          }
67      }
68  
69      /**
70       * Sets the specified value into the specified field while wrapping checked exception in a {@link
71       * RoseException}.
72       *
73       * @param field the field.
74       * @param self  the instance to set the field on.
75       * @param value the value to set.
76       */
77      public static void setValue(Field field, Object self, Object value) {
78          try {
79              field.set(self, value);
80          } catch (IllegalAccessException e) {
81              throw RoseException.wrap(e, RoseErrorCode.UNABLE_TO_SET_FIELD)
82                  .put("field", field.toGenericString());
83          }
84      }
85  
86      /**
87       * Sets the specified field value while wrapping checked exception in a {@link RoseException}.
88       *
89       * @param field the field.
90       * @param self  the instance to get the value from.
91       * @param <T>   the type of the field value.
92       * @return the field value.
93       */
94      @SuppressWarnings("unchecked")
95      public static <T> T getValue(Field field, Object self) {
96          try {
97              return (T) field.get(self);
98          } catch (IllegalAccessException e) {
99              throw RoseException.wrap(e, RoseErrorCode.UNABLE_TO_GET_FIELD)
100                 .put("field", field.toGenericString());
101         }
102     }
103 }
