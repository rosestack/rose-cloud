16  package io.github.rose.xxljob.service.impl;
17  
18  import io.github.rose.core.exception.BusinessException;
19  import io.github.rose.xxljob.config.XxlJobProperties;
20  import io.github.rose.xxljob.model.XxlJobInfo;
21  import io.github.rose.xxljob.model.XxlJobInfoPage;
22  import io.github.rose.xxljob.model.XxlRestResponse;
23  import io.github.rose.xxljob.service.JobInfoService;
24  import io.github.rose.xxljob.service.JobLoginService;
25  import org.springframework.http.HttpEntity;
26  import org.springframework.http.HttpHeaders;
27  import org.springframework.http.MediaType;
28  import org.springframework.http.ResponseEntity;
29  import org.springframework.util.LinkedMultiValueMap;
30  import org.springframework.util.MultiValueMap;
31  import org.springframework.util.ReflectionUtils;
32  import org.springframework.web.client.RestTemplate;
33  
34  import java.lang.reflect.Field;
35  import java.util.List;
36  
37  public class JobInfoServiceImpl implements JobInfoService {
38  
39      private final JobLoginService jobLoginService;
40  
41      private final RestTemplate restTemplate;
42  
43      private final XxlJobProperties xxlJobProperties;
44  
45      public JobInfoServiceImpl(
46          JobLoginService jobLoginService, RestTemplate restTemplate, XxlJobProperties xxlJobProperties) {
47          this.jobLoginService = jobLoginService;
48          this.restTemplate = restTemplate;
49          this.xxlJobProperties = xxlJobProperties;
50      }
51  
52      public static MultiValueMap<String, Object> convertObjectToMultiValueMap(Object obj) {
53          MultiValueMap<String, Object> multiValueMap = new LinkedMultiValueMap<>();
54          Class<?> clazz = obj.getClass();
55          for (Field field : clazz.getDeclaredFields()) {
56              ReflectionUtils.makeAccessible(field);
57              String fieldName = field.getName();
58              Object fieldValue = null;
59              try {
60                  fieldValue = field.get(obj);
61              } catch (IllegalAccessException e) {
62                  throw new RuntimeException(e);
63              }
64              if (fieldValue != null) {
65                  multiValueMap.add(fieldName, fieldValue.toString());
66              }
67          }
68          return multiValueMap;
69      }
70  
71      @Override
72      public List<XxlJobInfo> listJob(Integer jobGroupId, String executorHandler) {
73          String url = xxlJobProperties.getAdmin().getAddresses() + "/jobinfo/pageList";
74  
75          MultiValueMap<String, Object> map = new LinkedMultiValueMap<>();
76          map.add("jobGroup", jobGroupId);
77          map.add("executorHandler", executorHandler);
78          map.add("triggerStatus", -1);
79  
80          HttpHeaders headers = new HttpHeaders();
81          headers.set("Cookie", jobLoginService.getCookie());
82  
83          HttpEntity<MultiValueMap<String, Object>> requestEntity = new HttpEntity<>(map, headers);
84          ResponseEntity<XxlJobInfoPage> response = restTemplate.postForEntity(url, requestEntity, XxlJobInfoPage.class);
85          return response.getBody().getData();
86      }
87  
88      @Override
89      public Integer addJob(XxlJobInfo xxlJobInfo) {
90          return Integer.valueOf(
91              executeAction("add", convertObjectToMultiValueMap(xxlJobInfo)).getContent());
92      }
93  
94      @Override
95      public void updateJob(XxlJobInfo xxlJobInfo) {
96          executeAction("update", convertObjectToMultiValueMap(xxlJobInfo));
97      }
98  
99      @Override
100     public void removeJob(Integer jobId) {
101         MultiValueMap<String, Object> params = new LinkedMultiValueMap<>();
102         params.add("id", jobId);
103         executeAction("remove", params);
104     }
105 
106     @Override
107     public void startJob(Integer jobId) {
108         MultiValueMap<String, Object> params = new LinkedMultiValueMap<>();
109         params.add("id", jobId);
110         executeAction("start", params);
111     }
112 
113     @Override
114     public void stopJob(Integer jobId) {
115         MultiValueMap<String, Object> params = new LinkedMultiValueMap<>();
116         params.add("id", jobId);
117         executeAction("stop", params);
118     }
119 
120     private XxlRestResponse executeAction(String action, MultiValueMap<String, Object> params) {
121         String url = xxlJobProperties.getAdmin().getAddresses() + "/jobinfo/" + action;
122 
123         HttpHeaders headers = new HttpHeaders();
124         headers.setContentType(MediaType.APPLICATION_FORM_URLENCODED);
125         headers.set("Cookie", jobLoginService.getCookie());
126 
127         HttpEntity<MultiValueMap<String, Object>> requestEntity = new HttpEntity<>(params, headers);
128         ResponseEntity<XxlRestResponse> response =
129             restTemplate.postForEntity(url, requestEntity, XxlRestResponse.class);
130 
131         XxlRestResponse xxlRestResponse = response.getBody();
132         if (xxlRestResponse.getCode() != 200) {
133             throw new BusinessException(xxlRestResponse.getMsg());
134         }
135         return xxlRestResponse;
136     }
137 }
